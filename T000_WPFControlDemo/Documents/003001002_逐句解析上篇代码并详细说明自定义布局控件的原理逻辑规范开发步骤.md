```c#
// 自定义瀑布流面板（按列排列，高度自适应）
public class WaterfallPanel : Panel
{
    // 附加属性：指定子元素的列索引
    public static readonly DependencyProperty ColumnIndexProperty =
        DependencyProperty.RegisterAttached("ColumnIndex", typeof(int), typeof(WaterfallPanel), new PropertyMetadata(0));

    public static void SetColumnIndex(UIElement element, int value) => element.SetValue(ColumnIndexProperty, value);
    public static int GetColumnIndex(UIElement element) => (int)element.GetValue(ColumnIndexProperty);

    // 列数（依赖属性）
    public int ColumnCount
    {
        get => (int)GetValue(ColumnCountProperty);
        set => SetValue(ColumnCountProperty, value);
    }
    public static readonly DependencyProperty ColumnCountProperty =
        DependencyProperty.Register("ColumnCount", typeof(int), typeof(WaterfallPanel), new PropertyMetadata(2, OnColumnCountChanged));

    private static void OnColumnCountChanged(DependencyObject d, DependencyPropertyChangedEventArgs e)
    {
        ((WaterfallPanel)d).InvalidateMeasure(); // 列数变化，重新测量
    }

    // 测量阶段：计算每列的高度，确定面板总尺寸
    protected override Size MeasureOverride(Size availableSize)
    {
        // 初始化每列的高度
        double[] columnHeights = new double[ColumnCount];
        for (int i = 0; i < ColumnCount; i++) columnHeights[i] = 0;

        // 测量每个子元素
        foreach (UIElement child in Children)
        {
            if (child.Visibility == Visibility.Collapsed) continue;
            
            // 让子元素测量自己的期望尺寸
            child.Measure(availableSize);
            Size childDesired = child.DesiredSize;

            // 找到当前最矮的列
            int minColumn = 0;
            for (int i = 1; i < ColumnCount; i++)
            {
                if (columnHeights[i] < columnHeights[minColumn]) minColumn = i;
            }

            // 记录列高度
            SetColumnIndex(child, minColumn);
            columnHeights[minColumn] += childDesired.Height;
        }

        // 面板的总宽度=可用宽度，总高度=最高列的高度
        double totalWidth = availableSize.Width;
        double totalHeight = columnHeights.Max();
        return new Size(totalWidth, totalHeight);
    }

    // 排列阶段：确定每个子元素的最终位置
    protected override Size ArrangeOverride(Size finalSize)
    {
        // 初始化每列的顶部位置
        double[] columnTop = new double[ColumnCount];
        double columnWidth = finalSize.Width / ColumnCount;

        foreach (UIElement child in Children)
        {
            if (child.Visibility == Visibility.Collapsed) continue;
            
            // 获取子元素的列索引
            int columnIndex = GetColumnIndex(child);
            // 计算子元素的位置和大小
            Rect arrangeRect = new Rect(
                columnIndex * columnWidth, // 左坐标
                columnTop[columnIndex],    // 上坐标
                columnWidth,               // 宽度
                child.DesiredSize.Height   // 高度
            );
            // 排列子元素
            child.Arrange(arrangeRect);
            // 更新列的顶部位置
            columnTop[columnIndex] += child.DesiredSize.Height;
        }

        return finalSize;
    }
}
```

​	这份代码是**WPF 自定义布局面板**的标准实现，核心是继承`Panel`基类，重写 WPF 布局的核心方法`MeasureOverride`/`ArrangeOverride`，实现**自动分配到最短列**的瀑布流效果。

#### 逐句解析：

1. **附加属性**：WPF 布局面板的**专属设计** → 属性由`WaterfallPanel`定义，但**挂载到子元素**上（比如给 Button 设置`ColumnIndex`）；
2. `RegisterAttached`：注册**附加依赖属性**（普通属性无法被 WPF 布局系统识别）；
3. 静态`Set/Get`方法：附加属性的**标准规范**，外部才能给子元素赋值 / 取值；
4. 作用：**记录每个子元素应该放在第几列**，是测量 + 排列的核心依据。

### 模块 4：重写 `MeasureOverride`（布局第一步：测量）

csharp:

```c#
// 测量阶段：计算每列的高度，确定面板总尺寸
protected override Size MeasureOverride(Size availableSize)
{
    // 1. 初始化数组：记录每一列的总高度
    double[] columnHeights = new double[ColumnCount];
    for (int i = 0; i < ColumnCount; i++) columnHeights[i] = 0;

    // 2. 遍历所有子元素，逐个测量
    foreach (UIElement child in Children)
    {
        // 规范：跳过折叠隐藏的元素（不参与布局）
        if (child.Visibility == Visibility.Collapsed) continue;
        
        // 核心：让子元素自己测量「期望尺寸」（必须调用！）
        child.Measure(availableSize);
        Size childDesired = child.DesiredSize;

        // 3. 找到【当前高度最矮】的列（瀑布流核心逻辑）
        int minColumn = 0;
        for (int i = 1; i < ColumnCount; i++)
        {
            if (columnHeights[i] < columnHeights[minColumn]) minColumn = i;
        }

        // 4. 给子元素标记列号，累加该列高度
        SetColumnIndex(child, minColumn);
        columnHeights[minColumn] += childDesired.Height;
    }

    // 5. 返回面板自身的总尺寸
    double totalWidth = availableSize.Width;
    double totalHeight = columnHeights.Max(); // 总高度=最高的那一列
    return new Size(totalWidth, totalHeight);
}
```

#### 核心解析（测量阶段的使命）：

✅ **WPF 布局第一阶段：测量**

​	父容器问子元素：**你需要多大空间？**

1. 初始化列高数组，记录每列总高度；
2. 必须调用`child.Measure()` → 子元素计算自己的宽高（`DesiredSize`）；
3. 瀑布流核心：**自动把元素放到最短的列**；
4. 最终返回面板的总大小（宽度 = 父容器给的宽度，高度 = 最高列高度）。

### 模块 5：重写 `ArrangeOverride`（布局第二步：排列）

csharp:

```c#
// 排列阶段：确定每个子元素的最终位置
protected override Size ArrangeOverride(Size finalSize)
{
    // 1. 初始化数组：记录每一列的「顶部起始坐标」
    double[] columnTop = new double[ColumnCount];
    // 2. 计算每一列的宽度（总宽度 ÷ 列数）
    double columnWidth = finalSize.Width / ColumnCount;

    // 3. 遍历子元素，逐个定位
    foreach (UIElement child in Children)
    {
        if (child.Visibility == Visibility.Collapsed) continue;
        
        // 4. 获取子元素的列号（测量阶段标记的）
        int columnIndex = GetColumnIndex(child);
        
        // 5. 计算子元素的最终位置和大小
        Rect arrangeRect = new Rect(
            columnIndex * columnWidth, // 左坐标：列号 × 列宽
            columnTop[columnIndex],    // 上坐标：列的顶部位置
            columnWidth,               // 宽度：固定列宽
            child.DesiredSize.Height   // 高度：子元素自身的期望高度
        );
        
        // 核心：把子元素放到指定位置（必须调用！）
        child.Arrange(arrangeRect);
        
        // 6. 更新列的顶部坐标（下一个元素接在下面）
        columnTop[columnIndex] += child.DesiredSize.Height;
    }

    return finalSize;
}
```

#### 核心解析（排列阶段的使命）：

✅ **WPF 布局第二阶段：排列**

​	父容器告诉子元素：**给你分配空间，你就待在这个位置**

1. 计算每列宽度、每列的起始 Y 坐标；
2. 根据测量阶段标记的`ColumnIndex`，给子元素定位；
3. 调用`child.Arrange()` → 完成子元素渲染定位；
4. 累加列高，实现**垂直堆叠**效果。

## 第二部分：自定义布局控件的**核心原理**

### 1. WPF 布局的底层规则：**两步布局法**

​	所有 WPF 布局（Grid/StackPanel/ 自定义面板）都遵循这个**铁律**：

```markdown
测量阶段 (MeasureOverride) → 排列阶段 (ArrangeOverride)
```

1. **测量**：只算大小，不算位置 → 确定「我需要多大空间」
2. **排列**：只定位置，不算大小 → 确定「元素放在哪里」

### 2. `Panel`基类的作用

​	`Panel`是所有布局控件的基类，它帮我们做了：

- 管理子元素集合 `Children`

- 对接 WPF 布局系统

- 提供可视化树渲染能力

  

  我们只需要重写 2 个方法，就能实现任意布局！

### 3. 瀑布流的核心原理

- 测量时：**自动分配子元素到最短列**
- 排列时：**按列垂直堆叠，宽度均分，高度自适应**

## 第三部分：自定义布局控件的**开发逻辑**

### 整体逻辑（一句话）

​	**继承 Panel → 定义布局参数（附加 / 依赖属性）→ 测量子元素 → 排列子元素**

### 细分逻辑

1. **需求分析**：要做瀑布流 → 多列、均分宽度、自动分配到最短列；

2. **属性设计**：

   - 附加属性：给子元素标记列号；
   - 依赖属性：给面板设置总列数；

3. **测量逻辑**：

   ​	遍历子元素 → 测量尺寸 → 找最短列 → 标记列号 → 计算总高度；

4. **排列逻辑**：

   ​	遍历子元素 → 取列号 → 计算坐标 → 定位元素；

5. **规范处理**：

   ​	跳过隐藏元素、属性变更刷新布局、调用必须的 Measure/Arrange。

## 第四部分：WPF 自定义布局控件 **规范开发步骤**（行业标准）

​	这是所有 WPF 自定义布局控件**通用、规范、可复用**的开发流程，严格遵循即可写出高质量控件：

### 步骤 1：确定基类 → 必须继承 `Panel`

​	所有布局控件**唯一基类**，禁止直接继承`FrameworkElement`。

csharp:

```c#
public class 自定义面板名 : Panel { }
```

### 步骤 2：定义「附加属性」（子元素布局参数）

​	布局面板**必须用附加属性**存储子元素的布局信息（如列号、行号、位置）：

1. 用`DependencyProperty.RegisterAttached`注册；
2. 编写静态`SetXXX/GetXXX`包装方法；
3. 设置默认值、变更回调。

### 步骤 3：定义「依赖属性」（面板自身配置）

​	面板的公开配置（如列数、间距、方向）：

1. 用`DependencyProperty.Register`注册；
2. 实现 CLR 包装属性（`get/set`）；
3. 属性变更时调用`InvalidateMeasure()`刷新布局。

### 步骤 4：重写 `MeasureOverride`（测量核心）

**必须遵守规范**：

1. 遍历`Children`，跳过`Visibility.Collapsed`元素；
2. **必须调用`child.Measure(availableSize)`**；
3. 计算子元素尺寸、面板总尺寸；
4. 返回面板的`Size`。

### 步骤 5：重写 `ArrangeOverride`（排列核心）

**必须遵守规范**：

1. 遍历`Children`，跳过隐藏元素；
2. 计算子元素的最终坐标`Rect`；
3. **必须调用`child.Arrange(arrangeRect)`**；
4. 返回最终尺寸。

### 步骤 6：测试与优化（规范收尾）

1. XAML 测试：绑定数据、修改列数、测试隐藏元素；
2. 边界处理：列数 = 1、子元素为空、父容器尺寸变化；
3. 性能优化：避免重复计算、减少循环。

## 第五部分：总结（核心必背）

### 1. 代码核心总结

- 附加属性`ColumnIndex`：**标记子元素列号**；
- 依赖属性`ColumnCount`：**配置总列数**；
- `MeasureOverride`：**算大小、分配列**；
- `ArrangeOverride`：**定位置、堆叠元素**。

### 2. 自定义布局控件核心

- 基类：`Panel`
- 核心方法：`MeasureOverride` + `ArrangeOverride`
- 核心规则：**先测量、后排列**
- 核心规范：必须调用`child.Measure()`/`child.Arrange()`

### 3. 开发步骤（固化记忆）

**继承 Panel → 加附加属性 → 加依赖属性 → 写测量 → 写排列 → 测试优化**。
